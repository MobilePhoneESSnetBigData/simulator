36 bool MobilePhone::tryConnectNaiveAlgorithm(HoldableAgent::CONNECTION_TYPE type) {
37  bool connected = false;
38 
39  //select the most powerful antenna
40  Point *p = this->getLocation();
41  bool use_power = (type == HoldableAgent::CONNECTION_TYPE::USING_POWER);
42  double threshold = (use_power ? m_powerThreshold : Constants::QUALITY_THRESHOLD);
43 
44  if (m_connectedTo != nullptr) {
45  //check if this antenna is still in range, otherwise detach from it
46  bool inRange = EMField::instance()->isAntennaInRange(p, m_connectedTo, threshold, use_power);
47  if (!inRange) {
48  m_connectedTo->dettachDevice(this);
49  m_connectedTo = nullptr;
50  connected = false;
51  }
52  }
53  pair<Antenna*, double> antenna;
54  if (use_power)
55  antenna = EMField::instance()->computeMaxPower(p);
56  else
57  antenna = EMField::instance()->computeMaxQuality(p);
58 
59 
60  if (antenna.second > threshold) {
61  connected = antenna.first->tryRegisterDevice(this);
62  }
63  if (connected) {
64  if (m_connectedTo != nullptr && antenna.first->getId() != m_connectedTo->getId()) {
65  m_connectedTo->dettachDevice(this);
66  }
67  m_connectedTo = antenna.first;
68  }
69  else {
70  //try to connect to another antenna in range
71  vector<pair<Antenna*, double>> antennas = EMField::instance()->getInRangeAntennas(p, threshold, use_power);
72  unsigned long size = antennas.size();
73  for (unsigned long i = 0; i < size; i++) {
74  connected = antennas[i].first->tryRegisterDevice(this);
75  if (connected) {
76  if (m_connectedTo != nullptr && antennas[i].first->getId() != m_connectedTo->getId()) {
77  m_connectedTo->dettachDevice(this);
78  }
79  //m_connectedTo->dettachDevice(this);
80  m_connectedTo = antennas[i].first;
81  break;
82  }
83  }
84  }
85  return (connected);
86 }
