26  vector<Point*> generatePoints(Map* m, int n) {
27  vector<Point*> result;
28 
29  Geometry* g = m->getBoundary();
30  if (dynamic_cast<Polygon*>(g) != nullptr) {
31  Polygon* p = dynamic_cast<Polygon*>(g);
32  const Envelope* env = p->getEnvelopeInternal();
33  double xmin = env->getMinX();
34  double xmax = env->getMaxX();
35  double ymin = env->getMinY();
36  double ymax = env->getMaxX();
37 
38  double* x = RandomNumberGenerator::instance()->generateUniformDouble(xmin, xmax, n);
39  double* y = RandomNumberGenerator::instance()->generateUniformDouble(ymin, ymax, n);
40  for (int i = 0; i < n; i++) {
41  Coordinate c = Coordinate(x[i], y[i]);
42  result.push_back(m->getGlobalFactory()->createPoint(c));
43  }
44  delete[] x;
45  delete[] y;
46  }
47  return (result);
48  }
