364 vector<Person*> World::generatePopulation(unsigned long numPersons, vector<double> params, Person::AgeDistributions age_distribution,
365  double male_share, double probMobilePhone, double speed_walk, double speed_car) {
366 
367  vector<Person*> result;
368 
369  unsigned long id;
370  vector<Point*> positions = utils::generatePoints(getMap(), numPersons);
371 
372  int* walk_car = RandomNumberGenerator::instance()->generateBinomialInt(1, 0.5, numPersons);
373  int* phone = RandomNumberGenerator::instance()->generateBinomialInt(1, probMobilePhone, numPersons);
374  int sum = 0;
375  unsigned long numPhones = 0;
376  for (unsigned long i = 0; i < numPersons; i++) {
377  sum += walk_car[i];
378  numPhones += phone[i];
379  }
380 
381  int* gender = RandomNumberGenerator::instance()->generateBinomialInt(1, male_share, numPersons);
382  double* speeds_walk = RandomNumberGenerator::instance()->generateNormalDouble(speed_walk, 0.1 * speed_walk, numPersons - sum);
383  double* speeds_car = RandomNumberGenerator::instance()->generateNormalDouble(speed_car, 0.1 * speed_car, sum);
384 
385  double* ages;
386  if (age_distribution == Person::AgeDistributions::UNIFORM)
387  ages = RandomNumberGenerator::instance()->generateUniformDouble(params[0], params[1], numPersons);
388  else if (age_distribution == Person::AgeDistributions::NORMAL) {
389  ages = RandomNumberGenerator::instance()->generateTruncatedNormalDouble(params[2], params[3], params[0], params[1], numPersons);
390  }
391  else {
392  ages = RandomNumberGenerator::instance()->generateUniformDouble(params[0], params[1], numPersons);
393  }
394 
395  unsigned long cars = 0;
396  unsigned long walks = 0;
397  vector<MobilePhone*> mobiles = generateMobilePhones(numPhones, m_connType);
398  unsigned long m_index = 0;
399  Person* p;
400  for (unsigned long i = 0; i < numPersons; i++) {
401  id = IDGenerator::instance()->next();
402  if (walk_car[i])
403  p = new Person(getMap(), id, positions[i], m_clock, speeds_car[cars++], ages[i],
404  gender[i] ? Person::Gender::MALE : Person::Gender::FEMALE);
405  else
406  p = new Person(getMap(), id, positions[i], m_clock, speeds_walk[walks++], ages[i],
407  gender[i] ? Person::Gender::MALE : Person::Gender::FEMALE);
408 
409  if (phone[i]) {
410  mobiles[m_index]->setHolder(p);
411  p->addDevice(typeid(MobilePhone).name(), mobiles[m_index]);
412  m_index++;
413  }
414  result.push_back(p);
415  }
416  delete[] speeds_walk;
417  delete[] speeds_car;
418  delete[] ages;
419  delete[] gender;
420  return (result);
