108 double* RandomNumberGenerator::generateTruncatedNormalDouble(const double a, const double b, const double m, const double sd,
109  const unsigned long n) {
110  double* result = new double[n];
111  normal_distribution<double>::param_type p1(m, sd);
112  m_normal_double_distribution.param(p1);
113  unsigned long i = 0;
114  while (i < n) {
115  result[i] = m_normal_double_distribution(m_generator);
116  if (result[i] >= a && result[i] <= b)
117  i++;
118  }
119  return (result);
120 }
