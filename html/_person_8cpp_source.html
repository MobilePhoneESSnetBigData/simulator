83 void Person::randomWalk() {
84  double theta = 0.0;
85  double x, y, newX, newY;
86  theta = RandomNumberGenerator::instance()->generateUniformDouble(0.0, 2 * utils::PI);
87  x = getLocation()->getCoordinate()->x;
88  y = getLocation()->getCoordinate()->y;
89  newX = x + getSpeed() * cos(theta);
90  newY = y + getSpeed() * sin(theta);
91  Geometry* g = getMap()->getBoundary();
92 
93  Coordinate c = Coordinate(newX, newY);
94  Point* pt = getMap()->getGlobalFactory()->createPoint(c);
95 
96  if (pt->within(g)) {
97  this->getMap()->getGlobalFactory()->destroyGeometry(getLocation());
98  setLocation(pt);
99  }
100  else {
101  CoordinateSequence* cl = new CoordinateArraySequence();
102  cl->add(Coordinate(x, y));
103  cl->add(Coordinate(newX, newY));
104 
105  LineString* ls = this->getMap()->getGlobalFactory()->createLineString(cl);
106  Geometry* intersect = ls->intersection(g);
107  Point* ptInt = dynamic_cast<Point*>(intersect);
108  if (ptInt) {
109  this->getMap()->getGlobalFactory()->destroyGeometry(getLocation());
110  setLocation(ptInt);
111  }
112  }
